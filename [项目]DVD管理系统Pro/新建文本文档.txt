use dvdmgr;
drop table if exists my_admin;
drop table if exists dvd;

create table my_admin(
	admin_id int auto_increment not null comment '管理员ID',
	admin_name varchar(50) not null comment '管理员账号',
	admin_pwd varchar(50) not null comment '管理员密码',
	constraint pk_admin_id primary key(admin_id)
);

create table dvd(
	dvd_id int auto_increment not null comment 'DvdID',
	dvd_name varchar(30) not null comment 'Dvd名字',
	constraint pk_dvd_id primary key (dvd_id)
);
